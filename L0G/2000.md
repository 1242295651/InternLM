```python
class Solution:
    def canConstruct(self, ransomNote: str, magazine: str) -> bool:
        dict_1 = {}
        for i in ransomNote:
            dict_1[i]=dict_1.get(i,0)+1
        dict_2 = {}
        for i in magazine:
            dict_2[i]=dict_2.get(i,0)+1

        for i in ransomNote:
            if i not in dict_2 or dict_1[i]>dict_2[i]:
                return False
        return True

```
